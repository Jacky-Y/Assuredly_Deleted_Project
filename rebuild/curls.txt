import requests

url = "http://127.0.0.1:6000/query"
headers = {
    "Content-Type": "application/json"
}
data = {
    "InfoID": "0c1d2e3f4g5h"
}

response = requests.post(url, headers=headers, json=data)

print(response.status_code)
print(response.json())
